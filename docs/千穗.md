    ];

    const basePath = "/images/models/千穗/";

    // 生成左侧缩略图
    const thumbContainer = document.getElementById('thumbnail-list');
    images.forEach((img, index) => {
      const div = document.createElement('div');
      div.className = `thumbnail cursor-pointer rounded-lg overflow-hidden ${index === 0 ? 'active' : ''}`;
      div.innerHTML = `<img src="${basePath}${img.file}" class="w-full aspect-[3/4] object-cover">`;
      div.onclick = () => showImage(index);
      thumbContainer.appendChild(div);
    });

    // 显示大图
    let currentIndex = 0;
    function showImage(index) {
      currentIndex = index;
      const img = images[index];
      document.getElementById('main-image').src = basePath + img.file;
      document.getElementById('image-info').textContent = img.name;

      // 更新左侧高亮
      document.querySelectorAll('.thumbnail').forEach((el, i) => {
        el.classList.toggle('active', i === index);
      });
    }

    // 生成右侧列表
    const rightList = document.getElementById('right-list');
    images.forEach((img, i) => {
      const div = document.createElement('div');
      div.className = "cursor-pointer hover:text-pink-300 flex items-center gap-2";
      div.innerHTML = `📸 ${img.name}`;
      div.onclick = () => showImage(i);
      rightList.appendChild(div);
    });

    // 默认显示第一张
    showImage(0);
  </script>
