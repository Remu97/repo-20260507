<script>
  const images = [
    {name: "魔丽千穗3期", file: "魔丽千穗3期_10001.jpg"},
    {name: "魔丽千穗4期", file: "魔丽千穗4期_10002.jpg"},
    {name: "魔丽千穗6期", file: "魔丽千穗6期_10003.jpg"},
    {name: "魔丽千穗19期", file: "魔丽千穗19期_10004.jpg"},
    {name: "魔丽千穗20期", file: "魔丽千穗20期_10005.jpg"},
    {name: "魔丽千穗定制", file: "魔丽千穗定制_10006.jpeg"},
    {name: "精舞嘉美1期", file: "精舞嘉美1期_10007.jpeg"},
    {name: "RM王妃4期", file: "RM王妃4期_10008.jpeg"},
    {name: "炫舞格格22期", file: "炫舞格格22期_10009.jpeg"},
    {name: "假面No65-NO6", file: "假面No65-NO6_10010.jpeg"}
  ];

  const base = "/images/models/千穗/";

  // 生成左侧缩略图
  const thumbContainer = document.getElementById('thumb-list');
  images.forEach((img, i) => {
    const div = document.createElement('div');
    div.className = `cursor-pointer rounded-xl overflow-hidden ${i===0 ? 'ring-4 ring-pink-500' : ''}`;
    div.innerHTML = `<img src="${base}${img.file}" class="w-full aspect-[3/4] object-cover">`;
    div.onclick = () => showImage(i);
    thumbContainer.appendChild(div);
  });

  function showImage(i) {
    const img = images[i];
    document.getElementById('main-img').src = base + img.file;
    document.getElementById('img-name').textContent = img.name;

    // 更新左侧高亮
    Array.from(thumbContainer.children).forEach((el, idx) => {
      el.classList.toggle('ring-4', idx === i);
      el.classList.toggle('ring-pink-500', idx === i);
    });
  }

  // 生成右侧列表
  const rightList = document.getElementById('right-list');
  images.forEach((img, i) => {
    const div = document.createElement('div');
    div.className = "cursor-pointer hover:text-pink-400 flex items-center gap-2 py-1";
    div.innerHTML = `📸 ${img.name}`;
    div.onclick = () => showImage(i);
    rightList.appendChild(div);
  });

  // 默认显示第一张
  showImage(0);
</script>
